---
title: Agents API
description: Run AI agents with streaming and human-in-the-loop support.
---

The Agents API allows you to run AI agents, stream their progress, and provide human input when needed.

## Types

```typescript
type AgentRunStatus =
  | "queued"
  | "initializing"
  | "running"
  | "awaiting_input"
  | "succeeded"
  | "failed"
  | "canceled"
  | "timed_out"

type TriggerType = "manual" | "scheduled" | "email" | "luma_webhook"

interface AgentRunInput {
  prompt: string
  context?: unknown
  idempotencyKey?: string
}

interface AgentRun {
  runId: string
  agentId: string
  status: AgentRunStatus
  createdAt: string
}

interface AgentRunResult {
  id: string
  agentId: string
  status: AgentRunStatus
  result?: unknown
  error?: unknown
  completedAt?: string | null
  totalTokens?: number | null
  totalCostCents?: number | null
  steps: AgentRunStep[]
}

interface AgentRunStep {
  stepNumber: number
  type: string
  name?: string
  output?: unknown
  durationMs?: number
}
```

## Run an Agent

Start an agent run with a prompt:

```typescript
const { data, error } = await client.agents.run("agent_abc123", {
  prompt: "Analyze the quarterly sales data and create a summary report",
  context: {
    quarter: "Q4",
    year: 2024
  },
  idempotencyKey: "q4-report-2024"
})

if (data) {
  console.log("Run started:", data.runId)
}
```

## Get Run Status

Check the current status of a run:

```typescript
const { data } = await client.agents.getRun("run_xyz789")

console.log("Status:", data?.status)
console.log("Tokens used:", data?.totalTokens)
```

## Get Run Result

Retrieve the full result including steps:

```typescript
const { data, status } = await client.agents.getRunResult("run_xyz789")

if (status === 202) {
  console.log("Agent still running...")
} else if (data) {
  console.log("Result:", data.result)
  console.log("Steps:", data.steps.length)
  console.log("Cost:", data.totalCostCents, "cents")
}
```

## Wait for Result

Poll until the agent completes:

```typescript
const result = await client.agents.waitForResult("run_xyz789", {
  maxAttempts: 120, // Default: 120
  intervalMs: 1000  // Default: 1000ms
})

if (result.status === "succeeded") {
  console.log("Output:", result.result)

  for (const step of result.steps) {
    console.log(`Step ${step.stepNumber}: ${step.name} (${step.durationMs}ms)`)
  }
}
```

## Stream Run Progress

Get real-time updates using Server-Sent Events:

```typescript
const eventSource = client.agents.streamRun("run_xyz789")

eventSource.onmessage = (event) => {
  const data = JSON.parse(event.data)

  switch (data.event) {
    case "status":
      console.log("Status update:", data.data)
      break
    case "step":
      console.log("Step completed:", data.data)
      break
    case "done":
      console.log("Agent finished")
      eventSource.close()
      break
  }
}

eventSource.onerror = (error) => {
  console.error("Stream error:", error)
  eventSource.close()
}
```

## Human-in-the-Loop

When an agent requires human input, its status becomes `awaiting_input`. Provide the input to continue:

```typescript
// Check if agent needs input
const { data: status } = await client.agents.getRun("run_xyz789")

if (status?.status === "awaiting_input") {
  // Provide the required input
  const { data } = await client.agents.provideInput("run_xyz789", {
    input: {
      approved: true,
      comments: "Looks good, proceed with the changes"
    }
  })

  if (data?.success) {
    console.log("Input provided, agent will continue")
  }
}
```

## Cancel a Run

Stop a running agent:

```typescript
const { data } = await client.agents.cancelRun("run_xyz789")

if (data?.success) {
  console.log("Run canceled")
}
```

## Full Example

```typescript
import { createClient } from "@agents-server/sdk"

const client = createClient(process.env.API_KEY!, {
  baseUrl: process.env.API_URL!
})

async function runAnalysis(agentId: string, prompt: string) {
  // Start the agent
  const { data: run, error } = await client.agents.run(agentId, { prompt })

  if (error) {
    throw new Error(error.error)
  }

  // Wait for completion
  const result = await client.agents.waitForResult(run.runId)

  if (result.status === "failed") {
    throw new Error(String(result.error))
  }

  return {
    output: result.result,
    steps: result.steps,
    cost: result.totalCostCents
  }
}

// Usage
const analysis = await runAnalysis(
  "agent_sales_analyzer",
  "Generate Q4 2024 sales report"
)
console.log(analysis)
```
